  <script>
    const diasSemana = [
      'Segunda-feira',
      'Terça-feira',
      'Quarta-feira',
      'Quinta-feira',
      'Sexta-feira',
      'Sábado',
      'Domingo'
    ];

    let shoppingItems = [];

    function criarCamposCardapio() {
      const container = document.getElementById('menuContainer');
      container.innerHTML = '';

      diasSemana.forEach((dia, index) => {
        const box = document.createElement('div');
        box.className = 'day-box';
        box.innerHTML = `
          <h3>${dia}</h3>
          <div class="field-group">
            <label for="cafe-${index}">Café da manhã</label>
            <input id="cafe-${index}" type="text" placeholder="Ex.: pão, café, frutas" />
          </div>
          <div class="field-group">
            <label for="almoco-${index}">Almoço</label>
            <input id="almoco-${index}" type="text" placeholder="Ex.: arroz, feijão, frango" />
          </div>
          <div class="field-group">
            <label for="lanche-${index}">Lanche</label>
            <input id="lanche-${index}" type="text" placeholder="Ex.: bolo, suco, fruta" />
          </div>
          <div class="field-group">
            <label for="jantar-${index}">Jantar</label>
            <input id="jantar-${index}" type="text" placeholder="Ex.: sopa, sanduíche, macarrão" />
          </div>
        `;
        container.appendChild(box);
      });
    }

    function addShoppingItem() {
      const itemInput = document.getElementById('itemName');
      const categoryInput = document.getElementById('itemCategory');
      const nome = itemInput.value.trim();
      const categoria = categoryInput.value;

      if (!nome) {
        alert('Digite o nome do item.');
        return;
      }

      shoppingItems.push({
        id: Date.now(),
        nome,
        categoria,
        comprado: false
      });

      itemInput.value = '';
      renderShoppingList();
    }

    function renderShoppingList() {
      const list = document.getElementById('shoppingList');
      const summary = document.getElementById('summaryBox');
      list.innerHTML = '';

      shoppingItems.forEach(item => {
        const li = document.createElement('li');
        li.innerHTML = `
          <div class="item-left">
            <input type="checkbox" ${item.comprado ? 'checked' : ''} onchange="toggleItem(${item.id})" />
            <span class="${item.comprado ? 'checked' : ''}">${item.nome}</span>
            <span class="item-category">${item.categoria}</span>
          </div>
          <button class="delete-btn" onclick="deleteItem(${item.id})">Excluir</button>
        `;
        list.appendChild(li);
      });

      const comprados = shoppingItems.filter(i => i.comprado).length;
      summary.textContent = `Total de itens: ${shoppingItems.length} | Comprados: ${comprados}`;
    }

    function toggleItem(id) {
      shoppingItems = shoppingItems.map(item =>
        item.id === id ? { ...item, comprado: !item.comprado } : item
      );
      renderShoppingList();
    }

    function deleteItem(id) {
      shoppingItems = shoppingItems.filter(item => item.id !== id);
      renderShoppingList();
    }

    function salvarDados() {
      const menuData = diasSemana.map((_, index) => ({
        cafe: document.getElementById(`cafe-${index}`).value,
        almoco: document.getElementById(`almoco-${index}`).value,
        lanche: document.getElementById(`lanche-${index}`).value,
        jantar: document.getElementById(`jantar-${index}`).value
      }));

      const dados = {
        menuData,
        shoppingItems,
        observacoes: document.getElementById('observacoes').value
      };

      localStorage.setItem('cardapioSemanalRose', JSON.stringify(dados));
      alert('Dados salvos no navegador com sucesso!');
    }

    function carregarDados() {
      const dadosSalvos = localStorage.getItem('cardapioSemanalRose');
      if (!dadosSalvos) return;

      const dados = JSON.parse(dadosSalvos);

      if (dados.menuData) {
        dados.menuData.forEach((dia, index) => {
          const cafe = document.getElementById(`cafe-${index}`);
          const almoco = document.getElementById(`almoco-${index}`);
          const lanche = document.getElementById(`lanche-${index}`);
          const jantar = document.getElementById(`jantar-${index}`);

          if (cafe) cafe.value = dia.cafe || '';
          if (almoco) almoco.value = dia.almoco || '';
          if (lanche) lanche.value = dia.lanche || '';
          if (jantar) jantar.value = dia.jantar || '';
        });
      }

      shoppingItems = dados.shoppingItems || [];
      document.getElementById('observacoes').value = dados.observacoes || '';
      renderShoppingList();
    }

    function limparTudo() {
      if (!confirm('Deseja realmente apagar todo o conteúdo da página?')) return;

      diasSemana.forEach((_, index) => {
        document.getElementById(`cafe-${index}`).value = '';
        document.getElementById(`almoco-${index}`).value = '';
        document.getElementById(`lanche-${index}`).value = '';
        document.getElementById(`jantar-${index}`).value = '';
      });

      document.getElementById('observacoes').value = '';
      shoppingItems = [];
      renderShoppingList();
      localStorage.removeItem('cardapioSemanalRose');
    }

    function carregarExemplo() {
      const exemplo = [
        ['Pão e café', 'Arroz, feijão e frango', 'Banana', 'Sopa de legumes'],
        ['Cuscuz e leite', 'Macarrão com carne moída', 'Iogurte', 'Omelete'],
        ['Pão de queijo', 'Arroz, feijão e peixe', 'Maçã', 'Sanduíche natural'],
        ['Vitamina de banana', 'Frango grelhado e salada', 'Biscoito e suco', 'Sopa'],
        ['Pão com queijo', 'Feijão tropeiro', 'Mamão', 'Tapioca'],
        ['Bolo e café', 'Lasanha', 'Fruta', 'Caldo'],
        ['Pão e leite', 'Almoço especial em família', 'Suco', 'Lanche leve']
      ];

      exemplo.forEach((dia, index) => {
        document.getElementById(`cafe-${index}`).value = dia[0];
        document.getElementById(`almoco-${index}`).value = dia[1];
        document.getElementById(`lanche-${index}`).value = dia[2];
        document.getElementById(`jantar-${index}`).value = dia[3];
      });

      document.getElementById('observacoes').value = 'Comprar frutas frescas, verificar itens da geladeira e planejar marmitas da semana.';

      shoppingItems = [
        { id: 1, nome: 'Tomate', categoria: 'Hortifruti', comprado: false },
        { id: 2, nome: 'Banana', categoria: 'Hortifruti', comprado: false },
        { id: 3, nome: 'Frango', categoria: 'Carnes', comprado: false },
        { id: 4, nome: 'Arroz', categoria: 'Mercearia', comprado: true },
        { id: 5, nome: 'Leite', categoria: 'Laticínios', comprado: false }
      ];

      renderShoppingList();
    }

    criarCamposCardapio();
    carregarDados();
