    const storageKey = "planejador_4ano_trimestres_v1";
    const subjects = ["Português", "Matemática", "Ciências", "Geografia", "História"];
    let currentTrimester = "1º Trimestre";
    let editingId = null;
    let state = loadState();

    function loadState() {
      const saved = localStorage.getItem(storageKey);
      if (saved) return JSON.parse(saved);
      return normalize(JSON.parse(JSON.stringify(initialData)));
    }

    function normalize(data) {
      Object.keys(data).forEach(trim => {
        subjects.forEach(subject => {
          data[trim][subject] = (data[trim][subject] || []).map(item => ({
            id: item.id || crypto.randomUUID(),
            className: item.className || "4º ano",
            ...item
          }));
        });
      });
      return data;
    }

    function saveState() {
      localStorage.setItem(storageKey, JSON.stringify(state));
    }

    function fillSelects() {
      const subjectSelect = document.getElementById("subject");
      const filterSubject = document.getElementById("filterSubject");
      subjectSelect.innerHTML = subjects.map(s => `<option value="${s}">${s}</option>`).join("");
      filterSubject.innerHTML = `<option value="">Todas</option>` + subjects.map(s => `<option value="${s}">${s}</option>`).join("");
    }

    function renderTabs() {
      const tabs = document.getElementById("tabs");
      tabs.innerHTML = ["1º Trimestre", "2º Trimestre", "3º Trimestre"].map(trim => `
        <button class="tab ${trim === currentTrimester ? "active" : ""}" onclick="setTrimester('${trim}')">${trim}</button>
      `).join("");
      document.getElementById("trimester").value = currentTrimester;
    }

    function setTrimester(trim) {
      currentTrimester = trim;
      renderTabs();
      renderSummary();
      renderBoard();
    }

    function renderSummary() {
      const items = subjects.flatMap(subject => state[currentTrimester][subject]);
      const total = items.length;
      const done = items.filter(i => i.status === "Concluído").length;
      const progress = items.filter(i => i.status === "Em andamento").length;
      const pending = items.filter(i => i.status === "Não iniciado").length;

      document.getElementById("summary").innerHTML = `
        <div class="mini blue"><h3>Trimestre atual</h3><strong>${currentTrimester}</strong></div>
        <div class="mini pink"><h3>Total de itens</h3><strong>${total}</strong></div>
        <div class="mini green"><h3>Concluídos</h3><strong>${done}</strong></div>
        <div class="mini orange"><h3>Em andamento</h3><strong>${progress}</strong></div>
      `;
    }

    function getFilters() {
      return {
        search: document.getElementById("search").value.trim().toLowerCase(),
        subject: document.getElementById("filterSubject").value,
        status: document.getElementById("filterStatus").value
      };
    }

    function renderBoard() {
      const filters = getFilters();
      const board = document.getElementById("board");
      const visibleSubjects = subjects.filter(s => !filters.subject || filters.subject === s);

      board.innerHTML = visibleSubjects.map(subject => {
        const items = state[currentTrimester][subject].filter(item => {
          const haystack = `${item.code} ${item.title} ${item.content} ${item.notes} ${item.className}`.toLowerCase();
          const searchOk = !filters.search || haystack.includes(filters.search);
          const statusOk = !filters.status || item.status === filters.status;
          return searchOk && statusOk;
        });

        return `
          <div class="subject-card">
            <h3>${subject}</h3>
            ${items.length ? items.map(item => `
              <div class="entry">
                <div class="entry-top">
                  <div>
                    ${item.code ? `<span class="code">${item.code}</span>` : ""}
                    <h4>${item.title}</h4>
                  </div>
                  <div class="entry-actions">
                    <button onclick="editItem('${currentTrimester}','${subject}','${item.id}')">Editar</button>
                    <button onclick="deleteItem('${currentTrimester}','${subject}','${item.id}')">Excluir</button>
                  </div>
                </div>
                <p><strong>Conteúdo:</strong> ${item.content || "-"}</p>
                <p><strong>Turma:</strong> ${item.className || "4º ano"}</p>
                <p><strong>Andamento:</strong> ${item.status}</p>
                ${item.notes ? `<p><strong>Observações:</strong> ${item.notes}</p>` : ""}
              </div>
            `).join("") : `<div class="empty">Nenhum item encontrado.</div>`}
          </div>
        `;
      }).join("");
    }

    function clearForm() {
      editingId = null;
      document.getElementById("subject").value = subjects[0];
      document.getElementById("trimester").value = currentTrimester;
      document.getElementById("skillCode").value = "";
      document.getElementById("title").value = "";
      document.getElementById("content").value = "";
      document.getElementById("status").value = "Não iniciado";
      document.getElementById("className").value = "4º ano";
      document.getElementById("notes").value = "";
    }

    function editItem(trim, subject, id) {
      const item = state[trim][subject].find(i => i.id === id);
      if (!item) return;
      editingId = id;
      currentTrimester = trim;
      renderTabs();
      document.getElementById("subject").value = subject;
      document.getElementById("trimester").value = trim;
      document.getElementById("skillCode").value = item.code || "";
      document.getElementById("title").value = item.title || "";
      document.getElementById("content").value = item.content || "";
      document.getElementById("status").value = item.status || "Não iniciado";
      document.getElementById("className").value = item.className || "4º ano";
      document.getElementById("notes").value = item.notes || "";
      window.scrollTo({ top: 0, behavior: "smooth" });
      renderSummary();
      renderBoard();
    }

    function deleteItem(trim, subject, id) {
      if (!confirm("Deseja excluir este item?")) return;
      state[trim][subject] = state[trim][subject].filter(i => i.id !== id);
      saveState();
      renderSummary();
      renderBoard();
    }

    function saveItem() {
      const subject = document.getElementById("subject").value;
      const trim = document.getElementById("trimester").value;
      const item = {
        id: editingId || crypto.randomUUID(),
        code: document.getElementById("skillCode").value.trim(),
        title: document.getElementById("title").value.trim() || "Sem título",
        content: document.getElementById("content").value.trim(),
        status: document.getElementById("status").value,
        className: document.getElementById("className").value.trim() || "4º ano",
        notes: document.getElementById("notes").value.trim()
      };

      if (editingId) {
        subjects.forEach(sub => {
          ["1º Trimestre", "2º Trimestre", "3º Trimestre"].forEach(tr => {
            const index = state[tr][sub].findIndex(i => i.id === editingId);
            if (index >= 0) state[tr][sub].splice(index, 1);
          });
        });
      }

      state[trim][subject].unshift(item);
      currentTrimester = trim;
      editingId = null;
      saveState();
      clearForm();
      renderTabs();
      renderSummary();
      renderBoard();
    }

    document.getElementById("saveBtn").addEventListener("click", saveItem);
    document.getElementById("clearBtn").addEventListener("click", clearForm);
    document.getElementById("search").addEventListener("input", renderBoard);
    document.getElementById("filterSubject").addEventListener("change", renderBoard);
    document.getElementById("filterStatus").addEventListener("change", renderBoard);
    document.getElementById("printBtn").addEventListener("click", () => window.print());
    document.getElementById("resetBtn").addEventListener("click", () => {
      if (!confirm("Deseja restaurar os dados iniciais? Isso apagará suas alterações salvas no navegador.")) return;
      state = normalize(JSON.parse(JSON.stringify(initialData)));
      saveState();
      clearForm();
      renderSummary();
      renderBoard();
    });
    document.getElementById("teacherName").addEventListener("input", (e) => {
      localStorage.setItem("planejador_4ano_professora", e.target.value);
    });
    document.getElementById("teacherName").value = localStorage.getItem("planejador_4ano_professora") || "";

    fillSelects();
    clearForm();
    renderTabs();
    renderSummary();
    renderBoard();
    window.setTrimester = setTrimester;
    window.editItem = editItem;
    window.deleteItem = deleteItem;
  </script>
